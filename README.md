# ImportHelper
   
All env variables must be added
  For example
  - DB_URL=jdbc:mysql://localhost:3306/mybase;
  - DB_USERNAME=root;
  - DB_PASSWORD=root;
  - DB_NAME=mybase;
  - DB_TABLE_NAME=my_table;
  - INSERT_SIZE=5;
  - IMPORT_DB_NAME=newdb;
  - IMPORT_DV_TABLE_NAME=importDbTable;
  - FILE_NAME=buffer.sql;
  - FILE_PATH=./
